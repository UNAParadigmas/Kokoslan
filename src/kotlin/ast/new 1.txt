package kokoslan.ast;
import java.util.*;
import java.io.*;


public class KoKoMULT: KoKoBiOperation (KoKoAst operator, KoKoAst left, KoKoAst right): super(operator, left, right){

	override
	fun eval: KoKoValue( ctx: KoKoContext ){
	   try {
		     KoKoNumValue lv = (KoKoNumValue)(left().eval(ctx))
			 KoKoNumValue rv = (KoKoNumValue)(right().eval(ctx))
	         return new KoKoNumValue(lv.getValue() * rv.getValue())
	   } catch (Exception e) {
			throw new KoKoEvalException(e.getMessage())
	   }
   }
	
}
